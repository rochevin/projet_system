CREATE TABLE users (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	name VARCHAR(20),
	first_name VARCHAR(20),
	mail VARCHAR(40)
);

CREATE TABLE rapps (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	file_name VARCHAR(50),
	local_path VARCHAR(150),
	dist_path VARCHAR(150)
);

CREATE TABLE strats (
	id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	id_user INTEGER NOT NULL,
	id_rapp INTEGER NOT NULL,
	periodicity VARCHAR(20),
	date DATE,
	FOREIGN KEY (id_user) REFERENCES users(id),
	FOREIGN KEY (id_rapp) REFERENCES rapps(id)
);
